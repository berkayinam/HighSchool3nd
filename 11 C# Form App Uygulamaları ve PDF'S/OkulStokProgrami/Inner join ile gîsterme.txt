ÝKÝ TABLOYU BÝRLEÞTÝRME


SELECT ana_yer_tablosu.Ana_Yer_Adi, yer_tablosu.Yer_Adi
FROM ana_yer_tablosu
INNER JOIN yer_tablosu ON ana_yer_tablosu.Ana_Yer_ID = yer_tablosu.Ana_Yer_ID



ÜÇ TABLOYU BÝRLEÞTÝRME KODU

Select ana_yer_tablosu.Ana_Yer_Adi,yer_tablosu.Yer_Adi,Malzeme_Adi,Malzeme_Adedi from okul_malzeme_tablosu 
join ana_yer_tablosu on okul_malzeme_tablosu.Ana_Yer_ID = ana_yer_tablosu.Ana_Yer_ID 
join yer_tablosu on yer_tablosu.Yer_ID = okul_malzeme_tablosu.Yer_ID



ÜÇ TABLONUN ÝÇERÝÐÝNÝ SANAL TABLODA TUTMA KODU

Create view okul_malzeme_view
as Select ID, ana_yer_tablosu.Ana_Yer_Adi,yer_tablosu.Yer_Adi,Malzeme_Adi,Malzeme_Adedi from okul_malzeme_tablosu 
join ana_yer_tablosu on okul_malzeme_tablosu.Ana_Yer_ID = ana_yer_tablosu.Ana_Yer_ID 
join yer_tablosu on yer_tablosu.Yer_ID = okul_malzeme_tablosu.Yer_ID
